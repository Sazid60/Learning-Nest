## ২. Database Schema Issues Found (SQL থেকে)

### ❌ Problems in `Gaspay.sql` (MySQL syntax — PostgreSQL এ চলবে না)
```sql
-- MySQL backtick syntax → PostgreSQL এ double quotes বা no quotes
`tenants` → tenants  (fix: remove backticks)

-- AUTO_INCREMENT → BIGSERIAL বা GENERATED BY DEFAULT AS IDENTITY
`id` bigint PRIMARY KEY AUTO_INCREMENT → id BIGSERIAL PRIMARY KEY

-- jsonb → PostgreSQL এ ঠিক আছে কিন্তু MySQL এ নেই
`features` jsonb → ✅ PostgreSQL এ valid

-- Missing columns found:
-- customers table তে `credit_balance` column নেই → billing এ দরকার
-- bills table তে `applied_credits` column নেই → due_amount calculation এ দরকার
-- flats table তে `updated_at` নেই → Prisma এ দরকার
-- branches table তে `updated_at` নেই
-- stores table তে `updated_at` নেই
-- inventory_items table তে `updated_at` নেই

-- Missing FKs:
-- transactions.reference_id → payments.id এর FK নেই
-- employees table এ tenant_id FK আছে কিন্তু branch_id NULL constraint নেই

-- Missing Unique Indexes:
-- (flat_id) WHERE is_deleted=false → meters table (one meter per flat)
-- (meter_id, reading_month) → meter_readings table
-- (flat_id, billing_month) → bills table
-- (role_id, permission_id) → role_permissions table
-- (building_id) → billing_settings table (one per building)
```

## ৬. Global Exception Filter

```typescript
// src/common/filters/global-exception.filter.ts

import {
  ExceptionFilter, Catch, ArgumentsHost,
  HttpException, HttpStatus, Logger
} from '@nestjs/common';
import { Response } from 'express';
import { PrismaClientKnownRequestError } from '@prisma/client/runtime/library';

@Catch()
export class GlobalExceptionFilter implements ExceptionFilter {
  private readonly logger = new Logger(GlobalExceptionFilter.name);

  catch(exception: unknown, host: ArgumentsHost) {
    const ctx    = host.switchToHttp();
    const res    = ctx.getResponse<Response>();
    const req    = ctx.getRequest();

    let statusCode = HttpStatus.INTERNAL_SERVER_ERROR;
    let message    = 'Internal server error';

    if (exception instanceof HttpException) {
      statusCode = exception.getStatus();
      const resp = exception.getResponse();
      message = typeof resp === 'string'
        ? resp
        : (resp as any).message || message;
      // class-validator array → join করো
      if (Array.isArray(message)) message = message.join(', ');
    }

    // Prisma specific errors
    else if (exception instanceof PrismaClientKnownRequestError) {
      if (exception.code === 'P2002') {
        statusCode = HttpStatus.CONFLICT;
        message = 'Duplicate entry. Record already exists.';
      } else if (exception.code === 'P2025') {
        statusCode = HttpStatus.NOT_FOUND;
        message = 'Record not found.';
      } else if (exception.code === 'P2003') {
        statusCode = HttpStatus.BAD_REQUEST;
        message = 'Foreign key constraint failed.';
      } else if (exception.code === 'P2014') {
        statusCode = HttpStatus.BAD_REQUEST;
        message = 'Relation violation. Check related records.';
      }
    }

    this.logger.error(`[${req.method}] ${req.url} → ${statusCode}: ${message}`);

    res.status(statusCode).json({
      statusCode,
      success: false,
      message,
      data: null,
    });
  }
}
```

## ৭. Tenant Middleware (Subdomain থেকে Tenant সনাক্ত)

```typescript
// src/common/middleware/tenant.middleware.ts

import { Injectable, NestMiddleware, NotFoundException } from '@nestjs/common';
import { Request, Response, NextFunction } from 'express';
import { PrismaService } from '../../prisma/prisma.service';

@Injectable()
export class TenantMiddleware implements NestMiddleware {
  constructor(private readonly prisma: PrismaService) {}

  async use(req: Request, res: Response, next: NextFunction) {
    const host = req.hostname; // e.g. "metro.gaspay.cc"
    
    // Local development bypass
    if (host === 'localhost' || host === '127.0.0.1') {
      // Header দিয়ে override করার সুবিধা dev এ
      const tenantId = req.headers['x-tenant-id'];
      if (tenantId) {
        const tenant = await this.prisma.tenant.findUnique({
          where: { id: BigInt(tenantId as string) },
        });
        if (!tenant) throw new NotFoundException('Tenant not found');
        (req as any).tenantId = tenant.id;
        (req as any).tenant   = tenant;
      }
      return next();
    }

    // Production: subdomain parse
    const parts     = host.split('.');
    const subdomain = parts.length >= 3 ? parts[0] : null;

    if (!subdomain) {
      (req as any).tenantId = null;
      return next();
    }

    // Tenant lookup by email domain বা company slug
    // (tenants table এ একটা `subdomain` column add করা best practice)
    const tenant = await this.prisma.tenant.findFirst({
      where: {
        email:     { endsWith: `@${subdomain}` },
        isDeleted: false, // যদি soft delete থাকে
      },
    });

    if (!tenant) throw new NotFoundException('Tenant workspace not found');
    if (tenant.status === 'SUSPENDED')
      throw new NotFoundException('Tenant account is suspended');

    (req as any).tenantId = tenant.id;
    (req as any).tenant   = tenant;
    next();
  }
}
```

```typescript
import { Injectable, CanActivate, ExecutionContext, ForbiddenException } from '@nestjs/common';
import { Reflector } from '@nestjs/core';
import { PERMISSIONS_KEY } from '../decorators/require-permissions.decorator';

@Injectable()
export class PermissionsGuard implements CanActivate {
  constructor(private reflector: Reflector) {}

  canActivate(ctx: ExecutionContext): boolean {
    const required = this.reflector.getAllAndOverride<string[]>(
      PERMISSIONS_KEY,
      [ctx.getHandler(), ctx.getClass()]
    );
    if (!required?.length) return true;

    const { user } = ctx.switchToHttp().getRequest();
    const userPerms: string[] = user?.permissionsJson ?? [];

    const hasAll = required.every(p => userPerms.includes(p));
    if (!hasAll)
      throw new ForbiddenException(
        `Missing permission(s): ${required.filter(p => !userPerms.includes(p)).join(', ')}`
      );
    return true;
  }
}
```

```typescript
  defaultSort = 'createdAt',
): PaginationResult => {
  const page      = Math.max(1, Number(opts.page)  || 1);
  const limit     = Math.min(100, Math.max(1, Number(opts.limit) || 9));
  const skip      = (page - 1) * limit;

  const sortOrder = (opts.sortOrder === 'asc' ? 'asc' : 'desc') as 'asc' | 'desc';
  return { page, limit, skip, sortBy, sortOrder };
```

## ১০. Billing Engine (Core Logic)

```typescript
// src/common/utils/billing.engine.ts

export interface ChargeConfigWithSlabs {
  id:          bigint;
  chargeName:  string;
  chargeMode:  string;   // UNIT | FIXED | PERCENTAGE
  applyOn:     string;   // CONSUMPTION | SUBTOTAL | TOTAL
  isLateFee:   boolean;
  isSlab:      boolean;
  orderNo:     number;
  value:       number;
  slabs?: Array<{
    fromUnit: number;
    toUnit:   number | null;
    rate:     number;
  }>;
}

export interface BillingPreviewItem {
  chargeConfigId: bigint;
  chargeName:     string;
  chargeType:     string;
  baseOn:         string;
  value:          number;
  calculatedAmount: number;
}

export const calculateBill = (
  consumption: number,
  charges: ChargeConfigWithSlabs[],
): { items: BillingPreviewItem[]; totalAmount: number } => {
  // order_no অনুযায়ী sort — critical!
  const sorted = [...charges].sort((a, b) => a.orderNo - b.orderNo);

  let subtotal    = 0;
  let totalAmount = 0;
  const items: BillingPreviewItem[] = [];

  for (const charge of sorted) {
    let calculatedAmount = 0;

    // ─── UNIT charge ───────────────────────────────
    if (charge.chargeMode === 'UNIT') {
      if (charge.isSlab && charge.slabs?.length) {
        // Tiered slab calculation
        let remaining = consumption;
        for (const slab of charge.slabs.sort((a, b) => a.fromUnit - b.fromUnit)) {
          if (remaining <= 0) break;
          const slabMax   = slab.toUnit != null
            ? slab.toUnit - slab.fromUnit
            : Infinity;
          const inSlab    = Math.min(remaining, slabMax);
          calculatedAmount += inSlab * Number(slab.rate);
          remaining -= inSlab;
        }
      } else {
        // Flat rate
        calculatedAmount = consumption * charge.value;
      }
      subtotal += calculatedAmount;
    }

    // ─── FIXED charge ──────────────────────────────
    else if (charge.chargeMode === 'FIXED') {
      calculatedAmount = charge.value;
      subtotal += calculatedAmount;
    }

    // ─── PERCENTAGE charge ─────────────────────────
    else if (charge.chargeMode === 'PERCENTAGE') {
      const base =
        charge.applyOn === 'CONSUMPTION' ? consumption :
        charge.applyOn === 'SUBTOTAL'    ? subtotal    :
        /* TOTAL */                         totalAmount;
      calculatedAmount = base * (charge.value / 100);
      // Percentage charges subtotal এ count হয় না সাধারণত
    }

    totalAmount += calculatedAmount;

    items.push({
      chargeConfigId:   charge.id,
      chargeName:       charge.chargeName,
      chargeType:       charge.chargeMode,
      baseOn:           charge.applyOn,
      value:            charge.value,
      calculatedAmount: Math.round(calculatedAmount * 100) / 100,
    });
  }

  return { items, totalAmount: Math.round(totalAmount * 100) / 100 };
};
```

```
1. ChargeTypeModule → CRUD (order_no critical)
2. ChargeConfigModule → CRUD + date range
3. ChargeSlabModule → bulk upsert (last slab toUnit=null)
4. ChargeConfigMappingModule → bind to building
   (validate: no duplicate active type per building)
```

### Phase 7 — Billing Engine
```
1. BillingPreview → in-memory calc (no DB write)
2. BillingGenerate → bulk (per flat, DB transaction)
   - fetch active charge configs for building
   - calculate via billing.engine.ts
   - fetch previous_due from last unpaid bill
   - apply customer credit_balance
   - generate sequential bill_number
   - create bill + bill_items in one transaction
   - enqueue SMS notification
3. BillCRUD → list, detail, correct, soft-delete
```

### Phase 8 — Payments
```
1. ManualPayment → cash/bank record
   - if amount == due → PAID, due_amount = 0
   - if amount < due → PARTIALLY_PAID
   - if amount > due → PAID + credit_balance += excess
   - auto-create Transaction (double-entry)
2. bKash initiate → bKash API call
3. bKash callback → webhook verify + payment logic
```

### Phase 9 — Inventory
```
1. SupplierModule
2. MaterialModule + MaterialPurchase
   (auto-calc: total_price, effective_unit_cost)
3. StoreModule + InventoryItemModule
4. InventoryStock
5. StockTransfer (DB transaction: decrement+increment atomically)
```

### Phase 10 — Notifications, Settings, Cron
```
1. Settings CRUD (billing, sms, email, general, notification)
2. NotificationQueue worker (poll PENDING → send)
3. Late Fee Cron (@Cron daily):
   - query: UNPAID bills past (due_date + grace_days)
   - apply late fee charge config
4. Low Stock Alert Cron
```

### Phase 11 — Audit Logs
```
- Prisma middleware বা NestJS interceptor এ
- AUTO capture: action, entity, old/new values, user_id, IP
```

## ১৩. Custom Partial Indexes (Prisma Migration SQL)

এই indexes Prisma schema তে declare করা যায় না — আলাদা migration SQL এ লিখতে হবে:

```sql
-- prisma/migrations/0002_custom_indexes/migration.sql

-- One active meter per flat (unique partial index)
CREATE UNIQUE INDEX "meters_flat_id_unique_active"
  ON "meters" ("flat_id")
  WHERE "is_deleted" = false;

-- One active customer mapping per flat
CREATE UNIQUE INDEX "customer_flat_mappings_flat_id_unique_active"
  ON "customer_flat_mappings" ("flat_id")
  WHERE "is_active" = true;

-- Performance indexes
CREATE INDEX "bills_tenant_id_status_idx" ON "bills" ("tenant_id", "status");
CREATE INDEX "bills_flat_id_billing_month_idx" ON "bills" ("flat_id", "billing_month");
CREATE INDEX "meter_readings_meter_id_reading_month_idx" ON "meter_readings" ("meter_id", "reading_month");
CREATE INDEX "payments_tenant_id_payment_date_idx" ON "payments" ("tenant_id", "payment_date");
CREATE INDEX "inventory_stock_item_location_idx" ON "inventory_stock" ("item_id", "location_id", "location_type");
CREATE INDEX "notification_queue_status_idx" ON "notification_queue" ("status") WHERE "status" = 'PENDING';
CREATE INDEX "audit_logs_tenant_entity_idx" ON "audit_logs" ("tenant_id", "entity_type", "entity_id");
```

## ১৪. Key Validation Rules Summary

| Entity | Rule | Error |
|---|---|---|
| Flat occupancy | `customer_flat_mappings` তে একসাথে max 1 active per flat | 400 Already occupied |
| Meter per flat | `meters` তে is_deleted=false হলে unique flat_id | 400 Meter exists |
| Reading per month | `(meter_id, reading_month)` unique | 409 Duplicate reading |
| Reading value | `current_reading >= previous_reading` | 400 Invalid reading |
| Bill per flat/month | `(flat_id, billing_month)` unique | 409 Bill exists |
| Charge per building | Same charge_type active twice in 1 building | 400 Duplicate charge |
| Plan limits | `max_branch/building/flats` count check | 403 Limit exceeded |
| Stock transfer | from_location must have sufficient quantity | 400 Insufficient stock |
| Delete meter | Cannot delete flat with active meter (RESTRICT) | 400 Has active meter |
| Delete role | Cannot if users assigned | 400 Users exist |
| Payment on PAID | Cannot pay already PAID bill | 400 Bill paid |
| Tenant isolation | ALL queries filter `tenant_id = req.user.tenantId` | 403 Unauthorized |
